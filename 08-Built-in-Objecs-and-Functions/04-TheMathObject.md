# The Math Object
```javascript
var pi = Math.PI;
console.log(pi); // 3.141592653589793

var e = Math.E;
console.log(e); // 2.718281828459045

var a = -3;
console.log(Math.abs(a)); // 3

var b = 1.27;
console.log(Math.round(b)); // 1
console.log(Math.ceil(b)); // 2 Up

var c = 1.99;
console.log(Math.floor(c)); // 1 Down

var d = 2;
console.log(Math.exp(d)); // 7.38905609893065

var f = Math.E;
console.log(Math.log(f)); // 1

console.log(Math.max(1, 100, 1000)); // 1000

console.log(Math.random()); // 0.9165118723946413

var rnd = Math.random() * 100 + 1;
console.log(rnd); // 46.097467376572524

var rnd = Math.floor(Math.random() * 100) + 1;
console.log(rnd); // 75

```
